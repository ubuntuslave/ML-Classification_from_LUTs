19 #
20 # Generate learning_driver rules as a big LUT where irrelevant values get an invalid value of -1
21 # Valid values for components are only nonnegative integers in the interval [0, 9].
22 # Here, each possible combination of values among the relevant components becomes a rule.
23 # Thus, a table of decision rules is generated using random learning_driver.
24 # The length of the decision table depends on the number of groups on the components and the total number of relevant components.
25 # For example, for groups of 2 (pairs) and only 3 relevant components, there exists (10/2)^3 = 5^3 = 125 decision rules.
26 #
27 # @param class_labels: list of nonnegative integers for allowed classes (categories)
28 # @param num_of_components: Total number of components (or features)
29 # @param relevant_components: A list of those relevant components in the rules and any data set generated employing the table of decision rules created here.
30 # @param num_of_groups: If set to 10, then grouping doesn't have an effect since every value is a singleton.
31 #
32 # @return: The numpy array of rules where the last column corresponds to learning_driver labels.
33 #
34 def generate_rules(class_labels, num_of_components, relevant_components, num_of_groups):
35  # Create a LUT of num_of_components by n_rules
36  n_rules = num_of_groups ** len(relevant_components)
37  features_LUT = np.zeros((n_rules, num_of_components), dtype='int8') - 1
38 
39  # Fill up the relevant LUT of features with indices on the respective places
40  indices = np.arange(n_rules)
41 
42  multiple = n_rules
43  # Break up the indices in to its digits
44  for feat in relevant_components:
45  features_LUT[:, feat] = (indices % multiple) // int(multiple / num_of_groups)
46  multiple = multiple / num_of_groups
47 
48  num_of_classes = len(class_labels)
49 
50  # WISH: use string labels:
51  # labels = np.ndarray((total_num_of_instances), dtype=np.dtype({'classlabel':('S1', 0)})) # , dtype=str)
52 
53  # Create an array of labels of total_num_of_instances length
54  classes = np.asarray(class_labels, dtype='int8')
55  labels = classes[np.random.randint(num_of_classes, size=(n_rules))]
56 
57  # Shuffle labels
58  np.random.shuffle(labels)
59 
60  # Initialize all values in dataset to -1
61  complete_LUT = np.zeros((n_rules, num_of_components + 1), dtype='int8') - 1
62  complete_LUT[:, :num_of_components] = features_LUT
63  complete_LUT[:, -1] = labels
64 
65  return complete_LUT

68 #
69 # Irrelevant components are filled up with random values from a uniform distribution
70 # Last, the generated instance vectors (measurement records or rows including their associated class label) of the data set are shuffled and returned as the final result.
71 #
72 # @param rules: Numpy ndarray of rules organized as a 2D table of components and classes (last column)
73 # @param relevant_components: The list of relevant components
74 # @param total_num_of_instances: The desired number of instances
75 # @param instances_percentages: The split (percentage-wise) of instances from each category (class)
76 # @param num_of_groups: If set to 10, then grouping doesn't have an effect because groups would have a single element each.
77 #
78 # @return: The generated data set as a numpy array in which the last column corresponds to learning_driver labels.
79 #
80 def generate_dataset(rules, relevant_components, total_num_of_instances, instances_percentages, num_of_groups):
81  low = 0 # Lowest valid measurement value
82  high = 9 # Highest valid measurement value
83 
84  num_of_components = rules.shape[1] - 1
85  dataset = np.ndarray((total_num_of_instances, num_of_components + 1), dtype='int8')
86 
87  rules_c0 = rules[np.where(rules[..., -1] == 0)]
88  rules_indices_c0 = np.random.randint(len(rules_c0), size=(instances_percentages[0] * total_num_of_instances / 100))
89  dataset[:len(rules_indices_c0)] = rules_c0[rules_indices_c0]
90 
91  rules_c1 = rules[np.where(rules[..., -1] == 1)]
92  rules_indices_c1 = np.random.randint(len(rules_c1), size=(instances_percentages[1] * total_num_of_instances / 100))
93  dataset[len(rules_indices_c0):] = rules_c1[rules_indices_c1]
94 
95  # WISH: done manually for now: as proof of concept!
96  if num_of_groups < 10:
97  for f in relevant_components:
98  # Butterfly approach:
99  for q in range(num_of_groups): # Random reassignment done in reverse order to avoid overwrites
100  target_idx = np.where(dataset[:, f] == q)[0]
101  dataset[target_idx, f] = dataset[target_idx, f] + num_of_groups * np.random.randint(2, size=len(target_idx))
102 
103  # Resolve for irrelevant features
104  for f in range(num_of_components):
105  if f not in relevant_components:
106  # Draw samples from a uniform distribution.
107  irrelevant_feature_data = np.random.uniform(low, high, size=total_num_of_instances)
108  dataset[:, f] = irrelevant_feature_data
109 
110  # Shuffle rows
111  np.random.shuffle(dataset)
112 
113  return dataset
