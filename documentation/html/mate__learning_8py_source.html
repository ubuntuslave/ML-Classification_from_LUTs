101  skimmed_list = []

108  print("done! Elapsed: %.9f s" % (t_elapsed))

247  # A decision rules are learned from the provided list of matings (non-overlapping groups) and the chosen components
248  #
249  # @param matings: An ordered list of indices used to group employing the map_based of choice set in the constructor.
250  # By default, the mapping is pair-wise ordered from left to right.

252  #
253  # @return: The 1-D table of decision rules extracted from frequency analysis out of the learning portion of the data set
254  #
255  def generate_decision_rules_LUT(self, matings, comp_subset):
256  k_comps = len(comp_subset)
257  learning_LUT_of_probs = self._generate_training_LUT(k_comps) # For relative statistics work
258 
259  addresses_0 = self.get_address(matings, self.feat_train_c0[:, comp_subset]) # Mapped addresses
260  address_LUT_0 = self.get_index_address_table(addresses_0)
261  addresses_1 = self.get_address(matings, self.feat_train_c1[:, comp_subset]) # Mapped addresses
262  address_LUT_1 = self.get_index_address_table(addresses_1)
263 
264  # Count per catagory addresses and fill the table
265  count_c0 = np.bincount(address_LUT_0[:, 0])
266  count_c1 = np.bincount(address_LUT_1[:, 0])
267 
268  learning_LUT_of_probs[:len(count_c0), 0] = count_c0
269  learning_LUT_of_probs[:len(count_c1), 1] = count_c1
270  # Normalize
271  learning_LUT_of_probs = learning_LUT_of_probs / self.N_train
272 
273  # Fill up decision rules LUT
274  decision_rules_LUT = np.argmax(learning_LUT_of_probs, axis=-1)
275  # WISH: For now, the LUT will get assigned zeros if there are any ties! which may appear biased towards class 0!

288  # @param class_test: The classification labels associated with comp_data_test if any.
289  # @param verbose: Indicates whether to enable message verbosity.
290  #
291  # @return: The normalized accuracy of correct classification computed according to the specified parameters.
292  #
293  def compute_prediction_accuracy(self, matings, decisions_LUT, relevant_components=None, comp_data_test=None, class_test=None, verbose=False):
294  if comp_data_test == None:
295  observations, labels_true = self.feat_data_test, self.class_test
296  else:
297  observations, labels_true = comp_data_test, class_test
298 
299  if relevant_components != None:
300  observations = observations[:, relevant_components]
301 
302  N = observations.shape[0]
303 
304  predictions = self.predict_category(matings, decisions_LUT, observations)
305  labels_true = labels_true.reshape(*predictions.shape)
306  errors = predictions - labels_true
307  correct_count = N - np.count_nonzero(errors)
308  prob_correct = correct_count / float(N)
309  if verbose:
310  print("Accuracy = %.2f%s" % (prob_correct * 100., "%"), "with matings:", matings)
311 
312  return prob_correct
